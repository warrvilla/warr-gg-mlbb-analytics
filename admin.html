-- New policy: own row OR admin email
CREATE POLICY "Users read own profile" ON public.profiles
  FOR SELECT USING (
    auth.uid() = id
    OR auth.email() = 'wrrenvillapando@gmail.com'
  );

-- Also ensure write policies exist
CREATE POLICY IF NOT EXISTS "Users insert own profile" ON public.profiles
  FOR INSERT WITH CHECK (auth.uid() = id);
CREATE POLICY IF NOT EXISTS "Users update own profile" ON public.profiles
  FOR UPDATE USING (auth.uid() = id) WITH CHECK (auth.uid() = id);

When Supabase UI won't let you delete a user

Some auth users can't be removed from the Supabase dashboard due to foreign key constraints. Run this SQL in your Supabase → SQL Editor to force-delete by UUID. This cascades to profiles, teams, and scrims automatically.

-- Replace USER_UUID with the actual auth.users.id DELETE FROM auth.users WHERE id = 'USER_UUID';
